         if (contracts == null || contracts.isEmpty()) {
             throw new IllegalArgumentException("Contracts list cannot be null or empty");

         List<Contract> validContracts = new ArrayList<>();

     private boolean isValidContract(Contract contract) {
         if (contract == null) {
